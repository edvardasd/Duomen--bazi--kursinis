CREATE TABLE  "APYGARDA" 
   ( "ID" NUMBER NOT NULL ENABLE, 
 "PAVADINIMAS" VARCHAR2(255) NOT NULL ENABLE, 
  CONSTRAINT "APYGARDA_PK" PRIMARY KEY ("ID") ENABLE
   )

#### SCRIPT 2 ####
   
CREATE OR REPLACE TRIGGER  "BI_APYGARDA" 
  before insert on "APYGARDA"               
  for each row  
begin   
    select "APYGARDA_SEQ".nextval into :NEW.ID from dual; 
end;
